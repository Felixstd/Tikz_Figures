\documentclass[border=2pt]{standalone}
\usepackage{pgfplots}
\begin{document}
\begin{tikzpicture}
    \begin{axis}[
        xlabel={X-axis (log scale)},
        ylabel={Y-axis (log scale)},
        xmode=log, % Apply logarithmic scale to x-axis
        ymode=log, % Apply logarithmic scale to y-axis
        log basis x=10, % Base 10 logarithm for x-axis
        log basis y=10, % Base 10 logarithm for y-axis
        grid=major, % Show major grid lines
        legend pos=north west
    ]
        % Plot data
        \addplot[color=blue, mark=*] coordinates {
            (1, 1)
            (10, 100)
            (100, 10000)
            (1000, 1000000)
        };
        \legend{Sample Data}
    \end{axis}
\end{tikzpicture}
